Quick Example

## Fibonacci function with memoization
let cache = {}

def fib(n):
    if n <= 1:
        return n
    end
    
    if has_key(cache, n):
        return cache[n]
    end
    
    let result = fib(n - 1) + fib(n - 2)
    cache[n] = result
    return result
end

print(fib(30))  # Fast execution!
